

substring
----
${variable:offset:length}
echo "$variable" | cut -c start-end
expr substr "$variable" start length

${variable:offset}
----

remove suffix
----
${variable%suffix}
${image%.jp*}
# remove after the last dot
${filename%.*}
# remove after the first dot
${filename%%.*}
----

replace
----
# replace first
${parameter/pattern/string}
# replace all
${parameter//pattern/string}
----
