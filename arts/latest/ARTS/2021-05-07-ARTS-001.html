package com.pohvii.leetcode.p00;

import org.junit.jupiter.api.Test;

import java.util.Arrays;
import java.util.HashMap;
import java.util.Map;

import static org.junit.jupiter.api.Assertions.assertEquals;

/**
 * @see <a href="https://leetcode.com/problems/two-sum/">1. Two Sum</a>
 * @easy
 * @author zhanglei
 * @date 2021/5/17
 */
class P0001 {

    int[] twoSum(int[] nums, int target) {
//        return twoSum_s1(nums, target);
        return twoSum_s2(nums, target);
    }

    @Test
    void test() {
        assertEquals("[0, 1]", Arrays.toString(twoSum(new int[]{2, 7, 11, 15}, 9)));
        assertEquals("[1, 2]", Arrays.toString(twoSum(new int[]{3, 2, 4}, 6)));
        assertEquals("[0, 1]", Arrays.toString(twoSum(new int[]{3, 3}, 6)));
    }
    
    int[] twoSum_s1(int[] nums, int target) {
        int[] temps = new int[nums.length];
        for (int i = 0; i < nums.length; i++) {
            for (int j = 0; j < i; j++) {
                if (temps[j] == nums[i]) {
                    return new int[]{j, i};
                }
            }
            temps[i] = target - nums[i];
        }
        return null;
    }

    int[] twoSum_s2(int[] nums, int target) {
        Map<Integer, Integer> temps = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (temps.containsKey(nums[i])) {
                return new int[]{temps.get(nums[i]), i};
            }
            temps.put(target - nums[i], i);
        }
        return null;
    }
}
